void() locktime_touch =
{
		self.solid = SOLID_NOT;
		if(other.raceTime)
			dprint("Illogic! Race time is already set!");
		other.raceTime=time;
		self.think = hurt_on;
		self.nextthink = time + 1;
		return;
};
